\section{Spring Data JPA CRUD}

\input{current-section.tex}

\begin{frame}
 \frametitle{Interface \code{Repository}}
 \begin{itemize}
  \item Interface ``marker'' (ne définit pas de méthode)
  \item Définie dans Spring Data Commons
 \end{itemize}

\end{frame}

\begin{frame}
 \frametitle{Interface \code{CrudRepository}}
 \begin{itemize}
  \item Hérite de \code{Repository}
  \item Définit les méthodes CRUD de base
  \begin{itemize}
   \item \code{save}, \code{findOne}, \code{count}, \code{findAll}, etc.
  \end{itemize}
  \item Définie dans Spring Data Commons
 \end{itemize}

\end{frame}

\begin{frame}
 \frametitle{Interface \code{PagingAndSortingRepository}}
 \begin{itemize}
  \item Hérite de \code{CrudRepository}
  \item Définit les méthodes de pagination et de tri
  \item Définie dans Spring Data Commons
 \end{itemize}

\end{frame}

\begin{frame}
 \frametitle{Interface \code{JpaRepository}}
 \begin{itemize}
  \item Hérite de \code{PagingAndSortingRepository}
  \item Permet d'utiliser des \code{List} plutôt que des \code{Iterable}
  \item Rajoute la notion de \code{flush}
  \item Définie dans Spring Data JPA
 \end{itemize}

\end{frame}

\begin{frame}
 \frametitle{Classe \code{SimpleJpaRepository}}
 \begin{itemize}
  \item Implémentation de \code{JpaRepository}
  \item Utilisée par défaut via \code{<jpa:repositories />}
  \item Le développeur ne manipule généralement pas cette classe
 \end{itemize}

\end{frame}
